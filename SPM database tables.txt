CREATE TABLE file (
fileId int not null AUTO_INCREMENT PRIMARY KEY ,
fileName varchar(100) ,
date varchar(150) ,
cs int,
cns int,
cts int,
ci int,
cr int,
tw int 
)ENGINE=INNODB;

CREATE TABLE `measure`.`records` (
  `recordId` INT NOT NULL AUTO_INCREMENT,
  `value` VARCHAR(255) NULL,
  `fileId` INT NOT NULL,
  PRIMARY KEY (`recordId`),
  INDEX `fileId_idx` (`fileId` ASC),
  CONSTRAINT `fileId`
    FOREIGN KEY (`fileId`)
    REFERENCES `measure`.`file` (`fileId`)
    ON DELETE CASCADE
    ON UPDATE CASCADE);
